Fix8 Project

Fix8 (Fixate) is an Open-Source GUI Tool for Working with Eye Tracking Data in Reading Tasks. Fix8 includes a novel semi-automated correction approach for eye tracking data in reading tasks. The proposed approach allows the user to collaborate with an algorithm to produce accurate corrections in less time without sacrificing accuracy.

This documentation is part of the Fix8 project.

  16def find_background_color(img):
  17    """Private function that identifies the background color of the image
  18    Parameters
  19    ----------
  20    img : PIL.Image
  21        a PIL (pillow fork) Image object
  22    Returns
  23    -------
  24    str
  25        the color of the background of the image
  26    """
  27
  28    img = img.convert("L")  # Convert to grayscale
  29    threshold = 80
  30    img = img.point(
  31        lambda x: 0 if x < threshold else 255, "1"
  32    )  # Apply threshold and convert to black and white
  33
  34    width, height = img.size
  35
  36    color_result = []
  37    box_size = min(width, height) // 20
  38
  39    # Move a tiny rectangle box to obtain most common color
  40    for x, y in zip(range(0, width, box_size), range(0, height, box_size)):
  41        box = (x, y, x + box_size, y + box_size)
  42        minimum, maximum = img.crop(box).getextrema()
  43        color_result.append(minimum)
  44        color_result.append(maximum)
  45
  46    # Analyze and determine the background color
  47    if color_result.count(255) > color_result.count(0):
  48        bg_color = "white"
  49    else:
  50        bg_color = "black"
  51
  52    return bg_color

  54def EMTK_find_aoi(image_file_name=None, img=None, level="sub-line", margin_height=4, margin_width=7):
  55    """Find Area of Interest in the given image and store the aoi attributes in a Pandas Dataframe
  56    Parameters
  57    ----------
  58    image : str
  59        filename for the image, e.g. "vehicle_java.jpg"
  60    image_path : str
  61        path for all images, e.g. "emip_dataset/stimuli/"
  62    img : PIL.Image, optional
  63        PIL.Image object if user chooses to input an PIL image object
  64    level : str, optional
  65        level of detection in AOIs, "line" for each line as an AOI or "sub-line" for each token as an AOI
  66    margin_height : int, optional
  67        marginal height when finding AOIs, use smaller number for tight text layout
  68    margin_width : int, optional
  69        marginal width when finding AOIs, use smaller number for tight text layout
  70    Returns
  71    -------
  72    pandas.DataFrame
  73        a pandas DataFrame of area of interest detected by the method
  74    """
  75
  76    if img is None:
  77        if image_file_name is None:
  78            return
  79        
  80        # img = Image.open(image_path + image).convert('1')
  81        img = Image.open(image_file_name)
  82        img = img.convert("L")  # Convert to grayscale
  83        threshold = 80
  84        img = img.point(
  85            lambda x: 0 if x < threshold else 255, "1"
  86        )  # Apply threshold and convert to black and white
  87
  88    else:
  89        img = img.convert("L")  # Convert to grayscale
  90        threshold = 80
  91        img = img.point(
  92            lambda x: 0 if x < threshold else 255, "1"
  93        )  # Apply threshold and convert to black and white
  94
  95    width, height = img.size
  96
  97    # Detect the background color
  98    bg_color = find_background_color(img)
  99    # print("bg_color: ", bg_color)
 100
 101    left, right = 0, width
 102
 103    vertical_result, upper_bounds, lower_bounds = [], [], []
 104
 105    # Move the detecting rectangle from the top to the bottom of the image
 106    for upper in range(height - margin_height):
 107        lower = upper + margin_height
 108
 109        box = (left, upper, right, lower)
 110        minimum, maximum = img.crop(box).getextrema()
 111
 112        if upper > 1:
 113            if bg_color == "black":
 114                if vertical_result[-1][3] == 0 and maximum == 255:
 115                    # Rectangle detects white color for the first time in a while -> Start of one line
 116                    upper_bounds.append(upper)
 117                if vertical_result[-1][3] == 255 and maximum == 0:
 118                    # Rectangle detects black color for the first time in a while -> End of one line
 119                    lower_bounds.append(lower)
 120            elif bg_color == "white":
 121                if vertical_result[-1][2] == 255 and minimum == 0:
 122                    # Rectangle detects black color for the first time in a while -> Start of one line
 123                    upper_bounds.append(upper)
 124                if vertical_result[-1][2] == 0 and minimum == 255:
 125                    # Rectangle detects white color for the first time in a while -> End of one line
 126                    lower_bounds.append(lower)
 127
 128        # Storing all detection result
 129        vertical_result.append([upper, lower, minimum, maximum])
 130
 131    final_result = []
 132
 133    line_count = 1
 134
 135    # Iterate through each line of code from detection
 136    for upper_bound, lower_bound in list(zip(upper_bounds, lower_bounds)):
 137        # Reset all temporary result for the next line
 138        horizontal_result, left_bounds, right_bounds = [], [], []
 139
 140        # Move the detecting rectangle from the left to the right of the image
 141        for left in range(width - margin_width):
 142            right = left + margin_width
 143
 144            box = (left, upper_bound, right, lower_bound)
 145            minimum, maximum = img.crop(box).getextrema()
 146
 147            if left > 1:
 148                if bg_color == "black":
 149                    if horizontal_result[-1][3] == 0 and maximum == 255:
 150                        # Rectangle detects black color for the first time in a while -> Start of one word
 151                        left_bounds.append(left)
 152                    if horizontal_result[-1][3] == 255 and maximum == 0:
 153                        # Rectangle detects white color for the first time in a while -> End of one word
 154                        right_bounds.append(right)
 155                elif bg_color == "white":
 156                    if horizontal_result[-1][2] == 255 and minimum == 0:
 157                        # Rectangle detects black color for the first time in a while -> Start of one word
 158                        left_bounds.append(left)
 159                    if horizontal_result[-1][2] == 0 and minimum == 255:
 160                        # Rectangle detects white color for the first time in a while -> End of one word
 161                        right_bounds.append(right)
 162
 163            # Storing all detection result
 164            horizontal_result.append([left, right, minimum, maximum])
 165
 166        if level == "sub-line":
 167            part_count = 1
 168
 169            for left, right in list(zip(left_bounds, right_bounds)):
 170                final_result.append(
 171                    [
 172                        "sub-line",
 173                        f"line {line_count} part {part_count}",
 174                        left,
 175                        upper_bound,
 176                        right,
 177                        lower_bound,
 178                    ]
 179                )
 180                part_count += 1
 181
 182        elif level == "line":
 183            final_result.append(
 184                [
 185                    "line",
 186                    f"line {line_count}",
 187                    left_bounds[0],
 188                    upper_bound,
 189                    right_bounds[-1],
 190                    lower_bound,
 191                ]
 192            )
 193
 194        line_count += 1
 195
 196    # Format pandas dataframe
 197    columns = ["kind", "name", "x", "y", "width", "height", "image"]
 198    aoi = pd.DataFrame(columns=columns)
 199
 200    for entry in final_result:
 201        kind, name, x, y, x0, y0 = entry
 202        width = x0 - x
 203        height = y0 - y
 204        image = image_file_name.split("/")[-1]
 205
 206        # For better visualization
 207        x += margin_width / 2
 208        width -= margin_width
 209
 210        value = [kind, name, x, y, width, height, image]
 211        dic = dict(zip(columns, value))
 212
 213        aoi = aoi.append(dic, ignore_index=True)
 214
 215    return aoi, bg_color

 437def overlap(fix, AOI, radius=25):
 438    """Checks if fixation is within radius distance or over an AOI. Returns True/False.
 439
 440    Parameters
 441    ----------
 442    fix : Fixation
 443        A single fixation in a trial being considered for overlapping with the AOI
 444
 445    AOI : pandas.DataFrame
 446        contains AOI #kind	name	x	y	width	height	local_id	image	token
 447
 448    radius : int, optional
 449        radius around AOI to consider fixations in it within the AOI.
 450        default is 25 pixel since the fixation filter groups samples within 25 pixels.
 451
 452    Returns
 453    -------
 454    bool
 455        whether it overlaps
 456    """
 457
 458    box_x = AOI.x - (radius / 2)
 459    box_y = AOI.y - (radius / 2)
 460    box_w = AOI.width + (radius / 2)
 461    box_h = AOI.height + (radius / 2)
 462
 463    return box_x <= fix[0] and fix[0] <= box_x + box_w and box_y <= fix[1] and fix[1] <= box_y + box_h
